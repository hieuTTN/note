DECLARE new_device_code VARCHAR(255);        -- Lấy năm hiện tại    SET current_year = YEAR(CURDATE());            SET @lastID = (SELECT id FROM device ORDER BY id DESC LIMIT 1);    IF @lastID IS NULL OR @lastID = '' THEN        SET @lastID = 0;    END IF;    SET @lastID = @lastID +1;    -- Lấy id của thiết bị mới thêm vào sử dụng LAST_INSERT_ID()    SET new_id = @lastID;        -- Kiểm tra nếu device_code đã được cung cấp từ bên ngoài    -- Tạo mã thiết bị mới với độ dài là 6 ký tự, bắt đầu bằng 'bm' và năm hiện tại    SET new_device_code = CONCAT('BM', current_year, LPAD(new_id, 6, '0'));            -- Set giá trị cho cột device_code    SET NEW.code = new_device_code;END//DELIMITER ;DELIMITER //CREATE TRIGGER set_device_code_update BEFORE UPDATE ON
